function doGet(e) { return ContentService.createTextOutput("Sistem Server Absensi Aktif."); } function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var fileUrl = "-"; // Default jika tidak ada file // Logika pembuatan file di Google Drive jika ada dokumen yang diunggah if (data.fileData && data.fileName) { var decodedData = Utilities.base64Decode(data.fileData); var blob = Utilities.newBlob(decodedData, data.mimeType, data.fileName); // Mencari atau membuat folder khusus bernama "Bukti Absensi" di Drive Anda var folders = DriveApp.getFoldersByName("Bukti Absensi"); var folder = folders.hasNext() ? folders.next() : DriveApp.createFolder("Bukti Absensi"); var file = folder.createFile(blob); // Membuat agar link file bisa dilihat oleh guru/admin file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); fileUrl = file.getUrl(); } var rowData = [ data.date, data.nis, data.name, data.class, data.tipe_absen, data.status, data.terlambat, // Kolom Terlambat data.keterangan, // Kolom Keterangan Izin/Sakit fileUrl // Kolom Link File Bukti ]; sheet.appendRow(rowData); return ContentService.createTextOutput(JSON.stringify({ "status": "success" })) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
